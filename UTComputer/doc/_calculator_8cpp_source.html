65 bool Calculator::eventFilter(QObject *obj, QEvent *event)
66 {
67  /* Remet la position du curseur au debut de la ligne de commande dès que l'utilisateur clique
68  sur la ligne de commande. */
69  if (event->type() == QEvent::MouseButtonPress) {
70  command->setCursorPosition(command->text().size());
71  return true;
72  }
73  // Filtre des événements correspondant à une touche du clavier
74  if (event->type() == QEvent::KeyPress) {
75  QKeyEvent* keyEvent = static_cast<QKeyEvent*>(event);
76  std::string commandStr = command->text().toStdString();
77  //On ignore les fléches gauche et droite.
78  if (keyEvent->key() == Qt::Key_Right || keyEvent->key() == Qt::Key_Left) return true;
79  //Restauration de la commande antérieure
80  if (keyEvent->key() == Qt::Key_Up) {
81  if(commandStr.size() == 0) { //On part de la fin s'il n'y a rien sur la ligne
82  commands_pos = commands.size();
83  }
84  if(commands_pos > 0) { //On continue depuis la position courante sinon
85  command->setText(commands.at(--commands_pos));
86  }
87  }
88  else if(keyEvent->key() == Qt::Key_Down) {
89  if(commands_pos < (int)(commands.size() - 1)) {
90  command->setText(commands.at(++commands_pos));
91  }
92  else command->setText(QString(""));
93  }
94  //Opérateur arithmétique (un caractère) et guillemets fermés s'ils existent et programmes fermés
95  if (OperatorManager::getInstance().isArithmeticOperator(keyEvent->text().toStdString())
96  && std::count(commandStr.begin(), commandStr.end(), '"') % 2 == 0
97  && (std::count(commandStr.begin(), commandStr.end(), '[') == std::count(commandStr.begin(), commandStr.end(), ']'))) {
98  command->setText(command->text()+ QString(" ") + keyEvent->text());
99  calculate();
100  return true;
101  }
102  //On déclenche le calcul si entrée
103  if (keyEvent->key() == Qt::Key_Return || keyEvent->key() == Qt::Key_Enter) {
104  calculate();
105  return true;
106  }
107  return false;
108  } else {
109  // Sinon standard
110  return QObject::eventFilter(obj, event);
111  }
