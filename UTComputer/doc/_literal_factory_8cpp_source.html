112  auto prog = std::make_shared<ProgramLiteral>();
113  std::string copy(s);
114  size_t first, last;
115  if (s.front() == '[' && s.back() == ']') {
116  //On efface les délimiteurs de programme
117  copy.erase(0, 1);
118  copy.erase(copy.size() - 1);
119 
120  //On trouve la preuve d'un sous-programme (s'il existe)
121  first = copy.find_first_of('[');
122  last = copy.find_last_of(']');
123 
124  //Cas où n'y a pas de sous-programme (feuille)
125  if (first == std::string::npos && last == std::string::npos) makeLeafProgram(copy, prog);
126 
127  int leftBracketPos = -1; //Position courante crochet gauche
128  int rightBracketPos = -1; //Position courante du crochet droit
129  while((leftBracketPos = copy.find('[', rightBracketPos + 1)) != std::string::npos) { //Tant qu'on trouve un début de sous-programme ultérieur
130  //On traite la feuille à gauche du sous programme s'il existe
131  makeLeafProgram(copy.substr(rightBracketPos + 1, leftBracketPos - rightBracketPos - 1), prog);
132  int leftBracketCount = 1, rightBracketCount = 0;
133  for(unsigned int i = leftBracketPos + 1; i < copy.size(); ++i) {
134  if(leftBracketCount == rightBracketCount) break; //On a trouvé un sous programme complet
135  if(copy.at(i) == '[') ++leftBracketCount;
136  else if(copy.at(i) == ']') {
137  ++rightBracketCount;
138  rightBracketPos = i; //Mise à jour de la position de la fin effective du programme
139  }
140  }
141 
142  //Traitement du sous-programme
143  if(leftBracketPos != std::string::npos && rightBracketPos != std::string::npos) {
144  prog->add(makeCompositeProgram(copy.substr(leftBracketPos, rightBracketPos - leftBracketPos + 1)));
145  }
146 
147  //Sous-programme malformé
148  else {
149  throw ParsingError(s, "Square bracket unclosed in subprogram.");
150  }
151  }
152  //On traite l'éventuelle feuille à la fin du dernier sous-programme
153  if(last != std::string::npos) {
