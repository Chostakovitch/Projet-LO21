106 void Manager::handleOperandLine(std::string command) {
107  if(command.size() == 0) return;
108 
109  int leftPos;
110  int rightPos = -1;
111  //Une expression ne contient pas de sous expression, on considère seulement les paires de guillemets
112  while((leftPos = command.find('"', rightPos + 1)) != std::string::npos) {
113  rightPos = command.find('"', leftPos + 1);
114  if(leftPos != std::string::npos && rightPos != std::string::npos && leftPos != rightPos) {
115  command.erase(std::remove(command.begin() + leftPos, command.begin() + rightPos, ' '), command.begin() + rightPos);
116  }
117  }
118  //On traite à part les programmes et le reste, les programmes doivent rester entiers (pas de split sur espaces).
119  leftPos = command.find_first_of('[');
120  rightPos = command.find_last_of(']');
121  std::vector<std::string> tokens;
122  //On a trouvé la preuve d'au moins un programme, on effectue la séparation
123  if(leftPos != std::string::npos && rightPos != std::string::npos) {
124  int leftBracketPos = -1; //Position courante crochet gauche
125  int rightBracketPos = -1; //Position courante du crochet droit
126  //Tant qu'on trouve un début de programme, on trouve la fin (on saute donc les sous-programmes)
127  while((leftBracketPos = command.find('[', rightBracketPos + 1)) != std::string::npos) {
128  //On traite l'expression à gauche du programme
129  std::istringstream leftiss(command.substr(rightBracketPos + 1, leftBracketPos - rightBracketPos - 1));
130  tokens.insert(tokens.end(), std::istream_iterator<std::string>{leftiss}, std::istream_iterator<std::string>{});
131  int leftBracketCount = 1, rightBracketCount = 0;
132  for(unsigned int i = leftBracketPos + 1; i < command.size(); ++i) {
133  if(leftBracketCount == rightBracketCount) break; //On a trouvé un programme complet
134  if(command.at(i) == '[') ++leftBracketCount;
135  else if(command.at(i) == ']') {
136  ++rightBracketCount;
137  rightBracketPos = i; //Mise à jour de la position de la fin effective du programme
138  }
139  }
140 
141  //Traitement du programme en lui-même
142  if(leftBracketPos != std::string::npos && rightBracketPos != std::string::npos) {
143  tokens.push_back(command.substr(leftBracketPos, rightBracketPos - leftBracketPos + 1));
144  }
145 
146  //Programme malformé
147  else {
148  throw ParsingError(command, "Square bracket unclosed in program.");
149  }
150  }
151  //On traite l'expression à la fin du dernier programme
152  if(rightPos != std::string::npos) {
153  std::istringstream rightiss(command.substr(rightPos + 1));
154  tokens.insert(tokens.end(), std::istream_iterator<std::string>{rightiss}, std::istream_iterator<std::string>{});
155  }
156  }
157  //Sinon, on splitte tous les espaces
158  else {
159  std::istringstream iss(command);
160  tokens.insert(tokens.end(), std::istream_iterator<std::string>{iss}, std::istream_iterator<std::string>{});
161  }
162  std::vector<std::shared_ptr<Operand>> ops;
163  for(auto& op : tokens) {
164  //On essaye de fabriquer une littérale
165  try {
166  ops.push_back(LiteralFactory::getInstance().makeLiteralFromString(op));
167  }
168  catch(UTException& e1) {
169  //Non-littérale, on tente d'identifier un opérateur
170  try {
171  ops.push_back(OperatorManager::getInstance().getOperator(op));
172  }
173  catch(UTException& e2) {
174  //Non-opérateur, on tente de trouver un identificateur
175  try {
176  auto id = getIdentifier(op);
177  //On pushe la littérale correspondant à l'identificateur.
178  ops.push_back(id);
179 
180  //Atome correspondant à un programme : on l'évalue.
181  if(std::dynamic_pointer_cast<ProgramLiteral>(id)) ops.push_back(OperatorManager::getInstance().getEvalOperator());
182  }
183  //Non-identificateur, on crée une littérale expression si possible.
184  catch(UTException& e3) {
185  //Atome bien formé, on l'encapsule dans une LiteralExpression.
186  if(Utility::isAtom(op)) {
187  ops.push_back(LiteralFactory::getInstance().makeLiteral(op));
188  }
189  //Impossible de construire une opérande, on le signale
190  else throw ParsingError(op, "Atom must start with uppercase and contain only digits and uppercase chars").add(e3).add(e2).add(e1);
191  }
192  }
193  }
194  }
195  //On évalue ensuite les opérandes
196  try {
197  eval(ops);
198  saveState();
199  }
200  //Si quelque chose s'est mal passé, on restaure la pile avant l'évaluation
201  catch(UTException& e) {
202  if (backup.size() > 0) restoreState(backup[currentState]);
203  throw e;
204  }
205 }

207 void Manager::eval(std::vector<std::shared_ptr<Operand>> operands) {
208  //Pour chaque opérande
209  for(auto operand : operands) {
210  //Si on trouve une littérale, on l'empile
211  if(auto lit = std::dynamic_pointer_cast<Literal>(operand)) pile.push(lit);
212 
213  //Si on trouve un opérateur, on déclenche l'évaluation
214  else if(auto op = std::dynamic_pointer_cast<Operator>(operand)) {
215  if(pile.size() < op->getArity()) {
216  std::ostringstream oss;
217  oss << "Not enought operands for this operator. ";
218  oss << op->getArity() << " operands needed, " << pile.size() << " given.";
219  throw OperationError(op, Arguments<std::shared_ptr<Literal>>(), oss.str());
220  }
221  //Construction des arguments
222  Arguments<std::shared_ptr<Literal>> args;
223  args.reserve(op->getArity());
224  for(unsigned int i = 0; i < op->getArity(); ++i) {
225  args.insert(args.begin(), pile.pop());
226  }
227  //On tente d'effectuer l'opération (si elle est implémentée et si les types correspondent)
228  try {
229  auto res = OperatorManager::getInstance().dispatchOperation(op, args);
230  lastop = op;
231  lastargs = args;
232  //On évalue le résultat (il pourrait contenir des opérateurs).
233  eval(res);
234  }
235  //Impossible d'effectuer l'opération : on restitue la pile avant l'opération en cours.
236  catch(UTException& e) {
237  throw OperationError(op, args, "Failed to apply operation").add(e);
238  }
239  }
240  }
241 }
