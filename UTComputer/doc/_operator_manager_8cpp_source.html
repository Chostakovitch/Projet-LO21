151 std::shared_ptr<ExpressionLiteral> OperatorManager::opExpression(std::shared_ptr<Operator> op, const Arguments<std::shared_ptr<ExpressionLiteral>>& args) const {
152  std::ostringstream oss;
153  auto op_symbol = std::dynamic_pointer_cast<SymbolicOperator>(op);
154  //Cas d'un opérateur fonction, on l'applique simplement
155  if(!op_symbol) {
156  oss << op->toString() << "(";
157  //Ajout des opérandes
158  for(auto it = args.begin(); it != args.end(); ++it) {
159  oss << (*it)->getExpression();
160  //Excepté pour la dernière itération
161  if(std::next(it) != args.end()) {
162  oss << ',';
163  }
164  }
165  oss << ")";
166  }
167  //Cas d'un opérateur symbolique, il faut vérifier la priorité des opérateurs présents dans les expressions
168  else {
169  //On récupère un vecteur de tous les opérateurs symboliques définis ayant une priorité inférieure à la priorité de l'opérateur courant
170  std::vector<std::shared_ptr<Operator>> res;
171  std::copy_if(operators.begin(), operators.end(), std::back_inserter(res), [&op_symbol](const std::shared_ptr<Operator> op) {
172  auto s_op = std::dynamic_pointer_cast<SymbolicOperator>(op);
173  return s_op && s_op->getPriority() < op_symbol->getPriority();
174  });
175 
176  //Par définition du caractère symbolique, on connaît déjà l'arité : 2. On récupère les éléments non-parenthésés des expressions
177  std::string left = Utility::getOutside(args.at(0)->getExpression(), '(', ')');
178  std::string right = Utility::getOutside(args.at(1)->getExpression(), '(', ')');
179 
180  //Indique s'il faut parenthéser les expressions opérandes
181  bool parenthizeLeft = false;
182  bool parenthizeRight = false;
183 
184  //On itère sur tous les opérateurs existants de priorité inférieure
185  for(auto it : res) {
186  std::string symbol = it->toString();
187  //A gauche ou à droite, s'il existe un opérateur non parenthésé de priorité inférieure, il faut mettre des parenthèses
188  if(left.find(symbol) != std::string::npos) parenthizeLeft = true;
189  if(right.find(symbol) != std::string::npos) parenthizeRight = true;
190  }
191 
192  //Construction de l'expression
193  if(parenthizeLeft) oss << '(';
194  oss << args.at(0)->getExpression();
195  if(parenthizeLeft) oss << ')';
196  oss << op->toString();
197  if(parenthizeRight) oss << '(';
198  oss << args.at(1)->getExpression();
199  if(parenthizeRight) oss << ')';
200  }
201  return std::make_shared<ExpressionLiteral>(oss.str());
202 }
