283 void WindowEditIdentifier::save() {
284  if (valueTextEdit->toPlainText().isEmpty()) messageError->setText("Please enter a value");
285  else {
286  try {
287  std::string value = valueTextEdit->toPlainText().toStdString();
288  std::replace(value.begin(), value.end(), '\n', ' ');
289  std::replace(value.begin(), value.end(), '\t', ' ');
290 
291  // Erase white leading and trailing white spaces
292  value.erase(value.begin(), std::find_if(value.begin(), value.end(), std::bind1st(std::not_equal_to<char>(), ' ')));
293  value.erase(std::find_if(value.rbegin(), value.rend(), std::bind1st(std::not_equal_to<char>(), ' ')).base(), value.end());
294 
295  Manager::getInstance().changeIdentifier(key, value);
296  Manager::getInstance().saveState();
297  emit needRefresh();
298  close();
299  } catch (UTException e) {
300  messageError->setText(QString::fromStdString(e.details()));
301  }
302  }
